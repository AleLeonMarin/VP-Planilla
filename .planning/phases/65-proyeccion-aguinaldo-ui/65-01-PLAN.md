```typescript
static async calculateAguinaldo(
  employeeId: number,
  year: number
): Promise<{ total: number; months: number; promedio: number }> {
  const periodStart = new Date(`${year - 1}-12-01`);
  const periodEnd = new Date(`${year}-11-30`);
  const payrolls = await prisma.vpg_payrolls.findMany({
    where: {
      payrolls_period_end: { gte: periodStart, lte: periodEnd },
      payrolls_status: { in: [PayrollStatus.APROBADA, PayrollStatus.PAGADA] }
    },
    include: {
      vpg_payroll_employee: {
        where: { payroll_employee_employee_id: employeeId }
      }
    }
  });
  let totalGross = 0;
  let monthsWithSalary = 0;
  for (const payroll of payrolls) {
    for (const emp of payroll.vpg_payroll_employee) {
      const gross = Number(emp.payroll_employee_gross_salary || 0);
      totalGross += gross;
      if (gross > 0) monthsWithSalary++;
    }
  }
  return { total: totalGross / 12, months: monthsWithSalary, promedio: ... };
}
```

    ```typescript
    import { prisma } from '../lib/prisma';
    import { AguinaldoAccrual, AguinaldoSummaryRow } from '../model/AguinaldoAccrual';

    export class AguinaldoService {
      /**
       * Calculate accumulated aguinaldo for an employee as of a specific date.
       * Based on Costa Rica Labor Code Article 196.
       * Period: December 1 of prior year → asOfDate (capped at Nov 30).
       * Formula: (sum of gross_salary from APROBADA/PAGADA payrolls in period) / 12
       *
       * @param employeeId - The employee ID
       * @param asOfDate - Reference date (default: today)
       * @returns Promise<AguinaldoAccrual>
       * @throws Error if DB query fails
       */
      static async calculateAccruedAguinaldo(
        employeeId: number,
        asOfDate: Date = new Date()
      ): Promise<AguinaldoAccrual> {
        // Determine fiscal year period (Dec 1 of prior year → Nov 30 of current year)
        const year = asOfDate.getFullYear();
        // If asOfDate is in December, period is Dec 1 of this year → Nov 30 of next year
        // Otherwise period is Dec 1 of last year → Nov 30 of this year
        const periodStartYear = asOfDate.getMonth() === 11 ? year : year - 1;
        const periodStart = new Date(`${periodStartYear}-12-01`);
        const periodEndMax = new Date(`${periodStartYear + 1}-11-30`);
        // Cap at asOfDate if before Nov 30
        const periodEnd = asOfDate < periodEndMax ? asOfDate : periodEndMax;

        const payrolls = await prisma.vpg_payrolls.findMany({
          where: {
            payrolls_period_start: { gte: periodStart },
            payrolls_period_end: { lte: periodEnd },
            payrolls_status: { in: ['APROBADA', 'PAGADA'] },
          },
          include: {
            vpg_payroll_employee: {
              where: { payroll_employee_employee_id: employeeId },
            },
          },
        });

        let totalGross = 0;
        let payrollsIncluded = 0;

        for (const payroll of payrolls) {
          for (const emp of payroll.vpg_payroll_employee) {
            const gross = Number(emp.payroll_employee_gross_salary || 0);
            totalGross += gross;
            if (gross > 0) payrollsIncluded++;
          }
        }

        const accrued = Math.round((totalGross / 12) * 100) / 100;

        // monthsCompleted: approximate months from periodStart to asOfDate
        const msElapsed = Math.max(0, asOfDate.getTime() - periodStart.getTime());
        const monthsCompleted = Math.round(msElapsed / (1000 * 60 * 60 * 24 * 30.44));

        // projectedAnnual: if we maintain this pace for the full 12 months
        const projectedAnnual = monthsCompleted > 0
          ? Math.round(((totalGross / monthsCompleted) * 12 / 12) * 100) / 100
          : 0;

        return {
          accrued,
          projectedAnnual,
          periodStart,
          periodEnd,
          monthsCompleted,
          payrollsIncluded,
        };
      }

      /**
       * Get per-employee aguinaldo summary for a specific payroll.
       * Shows how much each employee had accrued BEFORE this payroll, 
       * the contribution FROM this payroll, and the total after approval.
       *
       * @param payrollId - The payroll ID
       * @returns Promise<AguinaldoSummaryRow[]>
       * @throws Error if payroll not found
       */
      static async getAguinaldoSummaryForPayroll(
        payrollId: number
      ): Promise<AguinaldoSummaryRow[]> {
        const payroll = await prisma.vpg_payrolls.findUnique({
          where: { payrolls_id: payrollId },
          include: {
            vpg_payroll_employee: {
              include: { vpg_employees: true },
            },
          },
        });

        if (!payroll) throw new Error('Planilla no encontrada');

        const periodStart = payroll.payrolls_period_start;
        // Determine fiscal period start
        const pYear = periodStart.getFullYear();
        const fiscalPeriodStart = periodStart.getMonth() === 11
          ? new Date(`${pYear}-12-01`)
          : new Date(`${pYear - 1}-12-01`);
        const fiscalPeriodEnd = new Date(`${pYear}-11-30`);

        const rows: AguinaldoSummaryRow[] = [];

        for (const empRow of payroll.vpg_payroll_employee) {
          const employeeId = empRow.payroll_employee_employee_id;
          const employeeName = empRow.vpg_employees
            ? `${empRow.vpg_employees.employee_first_name} ${empRow.vpg_employees.employee_last_name}`
            : `Empleado ${employeeId}`;

          // Prior payrolls in same fiscal period (excluding this payroll)
          const priorPayrolls = await prisma.vpg_payrolls.findMany({
            where: {
              payrolls_id: { not: payrollId },
              payrolls_period_start: { gte: fiscalPeriodStart },
              payrolls_period_end: { lte: fiscalPeriodEnd },
              payrolls_status: { in: ['APROBADA', 'PAGADA'] },
            },
            include: {
              vpg_payroll_employee: {
                where: { payroll_employee_employee_id: employeeId },
              },
            },
          });

          let priorGross = 0;
          for (const p of priorPayrolls) {
            for (const e of p.vpg_payroll_employee) {
              priorGross += Number(e.payroll_employee_gross_salary || 0);
            }
          }

          const thisGross = Number(empRow.payroll_employee_gross_salary || 0);
          const accruedBeforeThisPayroll = Math.round((priorGross / 12) * 100) / 100;
          const thisPayrollContribution = Math.round((thisGross / 12) * 100) / 100;
          const totalAccruedWithThis = Math.round(((priorGross + thisGross) / 12) * 100) / 100;

          rows.push({
            employeeId,
            employeeName,
            accruedBeforeThisPayroll,
            thisPayrollContribution,
            totalAccruedWithThis,
          });
        }

        return rows;
      }
    }
    ```
