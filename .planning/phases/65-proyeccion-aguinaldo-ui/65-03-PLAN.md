    2. CREATE src/frontend/src/services/aguinaldoService.ts:
    ```typescript
    import { http } from '@/services/http';
    import { AguinaldoAccrual, AguinaldoSummaryRow } from '@/types/aguinaldo';

    export const aguinaldoService = {
      /**
       * Fetch accrued aguinaldo for an employee as of today.
       * @param employeeId - The employee ID
       * @param asOfDate - Optional reference date (ISO string, default: today)
       */
      getEmployeeAguinaldo: (employeeId: number, asOfDate?: string): Promise<AguinaldoAccrual> => {
        const query = asOfDate ? `?asOfDate=${encodeURIComponent(asOfDate)}` : '';
        return http.get<AguinaldoAccrual>(`/employees/${employeeId}/aguinaldo${query}`);
      },

      /**
       * Fetch per-employee aguinaldo summary for a payroll.
       * @param payrollId - The payroll ID
       */
      getPayrollAguinaldoSummary: (payrollId: number): Promise<AguinaldoSummaryRow[]> => {
        return http.get<AguinaldoSummaryRow[]>(`/payroll/${payrollId}/aguinaldo-summary`);
      },
    };
    ```

    3. CREATE src/frontend/src/hooks/useAguinaldo.ts:
    ```typescript
    'use client';

    import { useState, useCallback, useEffect } from 'react';
    import { aguinaldoService } from '@/services/aguinaldoService';
    import { AguinaldoAccrual } from '@/types/aguinaldo';

    /**
     * Hook to fetch and manage aguinaldo accrual data for a single employee.
     * Returns { data, isLoading, error, refetch }.
     */
    export function useAguinaldo(employeeId: number | null) {
      const [data, setData] = useState<AguinaldoAccrual | null>(null);
      const [isLoading, setIsLoading] = useState(false);
      const [error, setError] = useState<string | null>(null);

      const fetchAguinaldo = useCallback(async () => {
        if (!employeeId) return;
        setIsLoading(true);
        setError(null);
        try {
          const result = await aguinaldoService.getEmployeeAguinaldo(employeeId);
          setData(result);
        } catch (err) {
          setError(err instanceof Error ? err.message : 'Error al obtener aguinaldo');
        } finally {
          setIsLoading(false);
        }
      }, [employeeId]);

      useEffect(() => {
        fetchAguinaldo();
      }, [fetchAguinaldo]);

      return { data, isLoading, error, refetch: fetchAguinaldo };
    }
    ```

        day: '2-digit',
        month: '2-digit',
        year: 'numeric',
      });

      const isEmpty = data.payrollsIncluded === 0;

      return (
        <div className="bg-white dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 border-l-4 border-l-[#4A5D3A] rounded-lg shadow-sm p-4">

              Aguinaldo acumulado
            </h3>
            <span className="text-xs text-zinc-500 dark:text-zinc-400">
              Acumulado a {today}
            </span>
          </div>

          {/* Amount */}
          <p className="text-xl font-bold text-zinc-800 dark:text-white mb-1">
            ₡{formatCRC(data.accrued)}
          </p>

          {isEmpty && (
            <p className="text-xs text-zinc-500 dark:text-zinc-400 mb-3">
              Sin planillas aprobadas en el período
            </p>

          {/* Progress bar */}
          <div className="mb-3">
            <div className="flex justify-between items-center mb-1">
              <span
