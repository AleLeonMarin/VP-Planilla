    ```typescript
    /**
     * Get aguinaldo accrual for an employee as of today (or a custom date).
     * GET /employees/:id/aguinaldo?asOfDate=2026-06-15
     * @param req - Express request (params.id = employeeId, query.asOfDate optional ISO string)
     * @param res - Express response with AguinaldoAccrual
     * @returns 200 { success: true, data: AguinaldoAccrual } | 400 | 500
     * @throws 400 if employeeId is not a positive integer
     */
    static async getEmployeeAguinaldo(req: Request, res: Response) {
      try {
        const employeeId = parseInt(req.params.id, 10);
        if (!Number.isInteger(employeeId) || employeeId <= 0) {
          return res.status(400).json({ success: false, error: 'employeeId debe ser un entero positivo' });
        }
        const asOfDate = req.query.asOfDate
          ? new Date(req.query.asOfDate as string)
          : new Date();
        const data = await AguinaldoService.calculateAccruedAguinaldo(employeeId, asOfDate);
        res.json({ success: true, data });
      } catch (error) {
        console.error('Failed to get employee aguinaldo:', error);
        res.status(500).json({ success: false, error: 'Failed to get employee aguinaldo' });
      }
    }

    /**
     * Get per-employee aguinaldo summary for a payroll.
     * GET /payroll/:id/aguinaldo-summary
     * @param req - Express request (params.id = payrollId)
     * @param res - Express response with AguinaldoSummaryRow[]
     * @returns 200 { success: true, data: AguinaldoSummaryRow[] } | 400 | 404 | 500
     * @throws 400 if payrollId is not a positive integer
     */
    static async getAguinaldoSummary(req: Request, res: Response) {
      try {
        const payrollId = parseInt(req.params.id, 10);
        if (!Number.isInteger(payrollId) || payrollId <= 0) {
          return res.status(400).json({ success: false, error: 'payrollId debe ser un entero positivo' });
        }
        const data = await AguinaldoService.getAguinaldoSummaryForPayroll(payrollId);
        res.json({ success: true, data });
      } catch (error) {
        const msg = error instanceof Error ? error.message : 'Failed to get aguinaldo summary';
        if (msg === 'Planilla no encontrada') {
          return res.status(404).json({ success: false, error: msg });
        }
        console.error('Failed to get aguinaldo summary:', error);
        res.status(500).json({ success: false, error: msg });
      }
    }
    ```
